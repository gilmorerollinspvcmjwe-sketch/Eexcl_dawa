### 1.3 关卡生成算法

```typescript
class LevelGenerator {
  // 根据难度等级生成关卡
  static generate(difficulty: DifficultyLevel): Level {
    const config = DIFFICULTY_PRESETS[difficulty];
    
    return {
      id: generateId(),
      name: config.name,
      description: config.description,
      difficulty,
      conditions: {
        minScore: config.baseScore * DIFFICULTY_MULTIPLIER[difficulty],
        minAccuracy: config.baseAccuracy,
        minCombo: config.baseCombo,
        maxTime: config.timeLimit,
      },
      targetConfig: {
        types: this.selectTargetTypes(difficulty),
        maxCount: config.maxTargets,
        spawnInterval: config.spawnInterval,
        duration: config.duration,
      },
      // 单格部位权重配置
      partWeights: this.getPartWeightsForDifficulty(difficulty),
      mechanics: this.selectMechanics(difficulty),
      excelConfig: this.generateExcelConfig(),
    };
  }
  
  // 根据难度获取部位生成权重
  static getPartWeightsForDifficulty(difficulty: number): PartWeights {
    const baseWeights: PartWeights = {
      head: 15,
      body: 35,
      leftHand: 18,
      rightHand: 18,
      foot: 14,
    };
    
    // 难度7以上增加头部比例
    if (difficulty >= 7) {
      baseWeights.head = Math.min(30, baseWeights.head + (difficulty - 6) * 3);
      baseWeights.foot = Math.max(5, baseWeights.foot - (difficulty - 6) * 2);
    }
    
    return baseWeights;
  }
  
  // 难度平衡公式
  static calculateDifficultyScore(level: Level): number {
    const targetScore = (
      level.targetConfig.maxCount * 10 +
      (level.targetConfig.spawnInterval[1] - level.targetConfig.spawnInterval[0]) * 20 +
      level.mechanics.length * 30 +
      level.difficulty * 15
    );
    
    return Math.round(targetScore * DIFFICULTY_MULTIPLIER[level.difficulty]);
  }
}
```

## 2. 单格部位射击系统

> **设计理念**：每个敌人仅占据一个单元格，显示单一部位图标。简化判定逻辑，提升游戏流畅度。

### 2.1 单格敌人数据结构

```typescript

interface SingleCellEnemy {
  id: string;
  position: {
    row: number;  // 所在行
    col: number;  // 所在列
  };
  bodyPart: BodyPartType;  // 显示的部位类型
  spawnTime: number;       // 出现时间戳

  | 'body'      // 身体 - 中等分值

  | 'rightHand' // 右手 - 较低分值
```

|------|------|----------|----------|----------|------|
| 头部 | 🎯 | 150分 | 3.0x | 高 | 爆头训练核心，最高回报 |

### 2.3 部位生成算法

```typescript
class EnemyGenerator {
  // 部位生成权重（可按关卡难度调整）
  private static BASE_WEIGHTS = {
    head: 15,      // 15% 概率生成头部
    body: 35,      // 35% 概率生成身体
    leftHand: 18,  // 18% 概率生成左手
    rightHand: 18, // 18% 概率生成右手
    foot: 14,      // 14% 概率生成脚部
  };
  
  // 根据难度调整权重
  static getWeightsForDifficulty(difficulty: number): PartWeights {
    const weights = { ...this.BASE_WEIGHTS };
    
    // 难度越高，头部比例越高
    if (difficulty >= 7) {
      weights.head = Math.min(30, weights.head + (difficulty - 6) * 3);
      weights.foot = Math.max(5, weights.foot - (difficulty - 6) * 2);
    }
    
    // 标准化权重
    const total = Object.values(weights).reduce((a, b) => a + b, 0);
    for (const key in weights) {
      weights[key as BodyPartType] = Math.round((weights[key as BodyPartType] / total) * 100);
    }
    
    return weights;
  }
  
  // 生成单个敌人
  static generateEnemy(
    gridRows: number, 
    gridCols: number, 
    difficulty: number,
    occupiedCells: Set<string>
  ): SingleCellEnemy | null {
    // 找到可用格子
    const availableCells = this.findAvailableCells(gridRows, gridCols, occupiedCells);
    if (availableCells.length === 0) return null;
    
    // 随机选择位置
    const { row, col } = availableCells[Math.floor(Math.random() * availableCells.length)];
    
    // 根据权重随机选择部位
    const weights = this.getWeightsForDifficulty(difficulty);
    const bodyPart = this.weightedRandomSelect(weights);
    
    return {
      id: generateId(),
      position: { row, col },
      bodyPart,
      spawnTime: Date.now(),
      duration: this.getDuration(difficulty),
      isHit: false,
    };
  }
  
  // 根据部位类型计算存活时长
  private static getDuration(difficulty: number): number {
    const baseDuration = 2000; // 2秒基础
    const difficultyReduction = Math.max(500, difficulty * 100); // 难度越高越短
    return baseDuration - difficultyReduction + Math.random() * 500;
  }
}
```

### 2.4 命中判定逻辑

```typescript
// 简化的命中判定 - 点击格子即命中该部位
function processHit(
  clickRow: number, 
  clickCol: number, 
  enemies: SingleCellEnemy[]
): HitResult {
  // 查找点击位置是否存在敌人
  const hitEnemy = enemies.find(
    e => e.position.row === clickRow && 
         e.position.col === clickCol &&
         !e.isHit
  );
  
  if (!hitEnemy) {
    return { 
      hit: false, 
      score: 0, 
      reason: 'miss',
      bodyPart: null 
    };
  }
  
  // 标记已命中
  hitEnemy.isHit = true;
  
  // 获取部位基础分值
  const baseScore = BODY_PART_SCORES[hitEnemy.bodyPart];
  
  // 计算连击加成
  const comboMultiplier = getComboMultiplier();
  
  // 最终得分
  const finalScore = Math.round(baseScore * comboMultiplier);
  
  return {
    hit: true,
    score: finalScore,
    bodyPart: hitEnemy.bodyPart,
    isHeadshot: hitEnemy.bodyPart === 'head',
    isBodyshot: hitEnemy.bodyPart === 'body',
    isLimbShot: ['leftHand', 'rightHand', 'foot'].includes(hitEnemy.bodyPart),
  };
}

// 部位分值映射
const BODY_PART_SCORES: Record<BodyPartType, number> = {
  head: 150,
  body: 100,
  leftHand: 60,
  rightHand: 60,
  foot: 40,
};
```

```typescript
// 连击倍率
function getComboMultiplier(): number {
  const combo = getCurrentCombo();
  
  if (combo >= 20) return 2.5;   // 20连击以上 2.5x
  if (combo >= 15) return 2.0;   // 15连击以上 2.0x
  if (combo >= 10) return 1.5;   // 10连击以上 1.5x
  if (combo >= 5)  return 1.2;   // 5连击以上 1.2x
  return 1.0;                    // 基础倍率
}

// 爆头额外奖励
function getHeadshotBonus(isHeadshot: boolean): number {
  return isHeadshot ? 50 : 0; // 爆头额外+50分
}

// 总分计算
function calculateTotalScore(hit: HitResult): number {
  const comboBonus = Math.round(hit.score * (getComboMultiplier() - 1));
  const headshotBonus = getHeadshotBonus(hit.isHeadshot);
  return hit.score + comboBonus + headshotBonus;
}
```

```typescript
function checkTrackingHit(
  clickPosition: Position,
  target: MovingTarget,
  deltaTime: number
): HitResult {
  // 预测位置（考虑移动延迟）
  const predictedPosition = predictTargetPosition(
    target.currentPosition,
    target.velocity,
    deltaTime * 2 // 2帧预测
  );
  
  // 计算与预测位置的距离
  const distance = calculateDistance(clickPosition, predictedPosition);
  
  // 判定有效命中
  if (distance < target.radius * 1.5) {
    // 距离越近，得分越高
    const accuracyBonus = Math.max(1.0, 2.0 - distance / target.radius);
    return {
      hit: true,
      score: BASE_SCORE * accuracyBonus,
      isTrackingHit: true,
    };
  }
  
  return { hit: false, score: 0 };
}
```

### 4.1 拐角射击模式（Peek Shot）

#### 核心机制

- 目标从单元格边缘"探出"，模拟 FPS 游戏中的 peek 动作
- 目标从左侧或右侧滑入，在中间区域停留短暂时间后滑出
- 停留窗口极短（0.3-0.8秒），考验反应速度

#### 参数配置

| 参数 | 范围 | 默认值 | 说明 |
|------|------|--------|------|
| 滑入速度 | 慢/中/快 | 中 | 目标进入速度 |
| 停留时间 | 0.3-1.0s | 0.5s | 有效命中窗口 |
| 生成间隔 | 1-5s | 2s | 目标出现间隔 |
| 滑出速度 | 与滑入相同 | - | 离开速度 |

#### 计分方式

| 命中时机 | 得分 | 说明 |
|----------|------|------|
| 停留窗口内 | 150分 | 有效命中 |
| 滑入/滑出时 | 0分 | 无效，不计分 |
| 超时 | miss | 目标消失 |

### 4.3 多目标切换模式
